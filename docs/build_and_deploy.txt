02. C:\CalypsoSystems\MatrixEase\MatrixEase.App\bin
03. C:\CalypsoSystems\MatrixEase\MatrixEase.App\obj
04. Rebuild Solution in release mode
05. Start window developer cmd prompt
06. cd C:\CalypsoSystems\MatrixEase\MatrixEase.App
07. electronize build /target win

10. copy "C:\CalypsoSystems\MatrixEase\MatrixEase.App\bin\Desktop\MatrixEase Setup 1.0.0.exe" "C:\transfer\MatrixEase Setup 1.0.0.exe"


Build Ubuntu Desktop 
=====================
01. Start "Clone of MatrixEase Manga Ubuntu 64-bit - Development" vm
02. Open a terminal
03. cd ~/dotnet/MatrixEase/
04. git pull
05. cd web_blaster
06. dotnet build --configuration Release
07. cd ../MatrixEase.App
08. sudo rm -r obj/
09. sudo rm -r bin/
10. electronize build /target linux
11. cp ~/dotnet/MatrixEase/MatrixEase.App/bin/Desktop/matrixease_1.0.0_amd64.snap ~/.
12. snapcraft login
13. snapcraft upload --release=beta ~/dotnet/MatrixEase/MatrixEase.App/bin/Desktop/matrixease_1.0.0_amd64.snap
14. copy ~/matrixease_1.0.0_amd64.snap to c:\transfer
15. from windows copy "C:\transfer\matrixease_1.0.0_amd64.snap" "C:\CalypsoSystems\MatrixEase\matrixease.com\download\matrixease_1.0.0_amd64.snap"


Build Docker Image
==================
01. cd ~/dotnet/MatrixEase/
02. git pull
03. cd web_blaster
04. dotnet build --configuration Release
05. cd ..
06. ~/dotnet/MatrixEase/bin/web_blaster Ubuntu Release ~/dotnet/MatrixEase/static.MatrixEase.Manga ~/dotnet/MatrixEase/matrixease.web/wwwroot googs
07. sudo docker build -t matrixease_web .
08. sudo docker rmi -f $(sudo docker images -f "dangling=true" -q)
11. sudo docker save matrixease_web > ~/matrixease_web.tar
12. copy ~/matrixease_web.tar to c:\transfer
15. from windows copy "C:\transfer\matrixease_web.tar" "C:\CalypsoSystems\MatrixEase\matrixease.com\download\matrixease_web.tar"


Init Docker Image
=================
01. cd ~/docker
02. ./docker_matrixease.sh stop
03. Update static.MatrixEase.Manga with all files
04. Update docs.matrixease.com with all files
05. copy matrixease_web.tar to /home/calypso/docker (on deployment machine)
06. sudo docker load <  matrixease_web.tar
07. sudo docker rmi -f $(sudo docker images -f "dangling=true" -q)
08. cd matrixease_dotnet
09. sudo docker-compose down
10. sudo docker-compose up -d
11. sudo docker-compose stop
12. cd ..
13. ./docker_matrixease.sh start
